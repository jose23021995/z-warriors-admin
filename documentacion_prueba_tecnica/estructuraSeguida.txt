para imprimir resultados en consola y en un txt 
function Show-Tree($path, $prefix = "") {
    $result = @()
    $items = Get-ChildItem $path | Sort-Object PSIsContainer -Descending, Name
    $count = $items.Count

    for ($i = 0; $i -lt $count; $i++) {
        $item = $items[$i]
        $isLast = ($i -eq $count - 1)

        $connector = if ($isLast) { "└── " } else { "├── " }
        $line = "$prefix$connector$($item.Name)"

        $result += $line

        if ($item.PSIsContainer) {
            $newPrefix = if ($isLast) { "$prefix    " } else { "$prefix│   " }
            $result += Show-Tree $item.FullName $newPrefix
        }
    }

    return $result
}
